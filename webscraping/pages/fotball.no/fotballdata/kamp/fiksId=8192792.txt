        var revealButtons = document.querySelectorAll('[id^="btn_rev_fetch_captcha_"]');
        revealButtons.forEach(function (revealButton) {
            revealButton.addEventListener('click', function () {
                var personId = this.getAttribute('data-person-id');
                var controllId = this.getAttribute('data-controll-id');

                // Initialize reCAPTCHA
                grecaptcha.ready(function () {
                    grecaptcha.execute('6LclITQoAAAAAEmMb_egyLrBd6xaTRm_s9H0FRo0', { action: 'reveal_info' }).then(function (token) {
                        // Required to send the recaptcha token to server for verification
                        fetch(`/Person/ValidateTokenAndGetPersonInfo?recaptchaToken=${token}&personId=${personId}`)
                            .then(
                                response => {
                                    // Parse and return the JSON response
                                    return response.json()
                                }
                            )
                            .then(data => {
                                var updatedEmail = false;
                                var updatedTel = false;

                                // Update the person information returned by the server
                                if (document.querySelector(`#email_${controllId}`) != null && data.EmailObfuscated && data.EmailObfuscated.length > 0) {
                                    document.getElementById(`email_${controllId}`).setAttribute("data-address", data.EmailObfuscated);
                                    updatedEmail = true;
                                }
                                if (document.querySelector(`#mobilephone_${controllId}`) != null && data.MobilePhoneObfuscated && data.MobilePhoneObfuscated.length > 0) {
                                    document.getElementById(`mobilephone_${controllId}`).setAttribute("data-address", data.MobilePhoneObfuscated);
                                    updatedTel = true;
                                }
                                if (document.querySelector(`#workphone_${controllId}`) != null && data.WorkPhoneObfuscated && data.WorkPhoneObfuscated.length > 0) {
                                    document.getElementById(`workphone_${controllId}`).setAttribute("data-address", data.WorkPhoneObfuscated);
                                    updatedTel = true;
                                }

                                // Hide the reveal button and show the person information
                                if (updatedEmail || updatedTel) {
                                    if (document.querySelector(`#infoContainer_${controllId}`) != null) {
                                        document.getElementById(`infoContainer_${controllId}`).style.display = 'block';
                                    }
                                    if (document.querySelector(`#btn_rev_fetch_captcha_${controllId}`) != null) {
                                        document.getElementById(`btn_rev_fetch_captcha_${controllId}`).style.display = 'none';
                                    }

                                    //Bind the email and telephone links
                                    if (updatedEmail) {
                                        BindAddressLinks('.address-link_' + controllId);
                                    }

                                    if (updatedTel) {
                                        BindTeleLinks('.tel-link_' + controllId);
                                    }
                                }
                            })
                            .catch((error) => {
                                console.error('Failed to verify reCAPTCHA or fetch the info. Please try again.');
                                console.error('Error:', error);
                            });
                    });
                });
            });
        });

        var revealButtonsOnlyCaptcha = document.querySelectorAll('[id^="btn_rev_captcha_"]');

        revealButtonsOnlyCaptcha.forEach(function (revealButton) {
            revealButton.addEventListener('click', function () {
                var controllId = this.getAttribute('data-controll-id');

                // Initialize reCAPTCHA
                grecaptcha.ready(function () {
                    grecaptcha.execute('6LclITQoAAAAAEmMb_egyLrBd6xaTRm_s9H0FRo0', { action: 'reveal_info' }).then(function (token) {
                        // Required to send the recaptcha token to server for verification
                        fetch(`/Person/ValidateToken?recaptchaToken=${token}`)
                            .then(
                                response => {
                                    // Parse and return the JSON response
                                    return response.json()
                                }
                            )
                            .then(data => {
                                // Hide the reveal button and show the person information
                                if (document.querySelector(`#infoContainer_${controllId}`) != null) {
                                    document.getElementById(`infoContainer_${controllId}`).style.display = 'block';
                                }

                                // Hides the clicked button
                                if (document.querySelector(`#btn_rev_captcha_${controllId}`) != null) {
                                    document.getElementById(`btn_rev_captcha_${controllId}`).style.display = 'none';
                                }
                                //Bind the email and telephone links
                                BindAddressLinks('.address-link_' + controllId);
                                BindTeleLinks('.tel-link_' + controllId);
                            })
                            .catch((error) => {
                                console.error('Failed to verify reCAPTCHA or fetch the info. Please try again.');
                                console.error('Error:', error);
                            });
                    });
                });
            });
        });
       
    };

    });

        addRecaptchaEventListeners();
