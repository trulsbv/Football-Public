    function addRecaptchaEventListeners() {
        //Adding event handlers to any reveal buttons.

        revealButtons.forEach(function (revealButton) {

                var personId = this.getAttribute('data-person-id');
                var controllId = this.getAttribute('data-controll-id');

                // Initialize reCAPTCHA
                grecaptcha.ready(function () {
                    grecaptcha.execute('6LclITQoAAAAAEmMb_egyLrBd6xaTRm_s9H0FRo0', { action: 'reveal_info' }).then(function (token) {
                        // Required to send the recaptcha token to server for verification
                        fetch(`/Person/ValidateTokenAndGetPersonInfo?recaptchaToken=${token}&personId=${personId}`)
                            .then(
                                response => {
                                    // Parse and return the JSON response
                                    return response.json()
                                }
                            )
                            .then(data => {
                                var updatedEmail = false;
                                var updatedTel = false;

                                // Update the person information returned by the server
                                if (document.querySelector(`#email_${controllId}`) != null && data.EmailObfuscated && data.EmailObfuscated.length > 0) {
                                    document.getElementById(`email_${controllId}`).setAttribute("data-address", data.EmailObfuscated);
                                    updatedEmail = true;
                                }
                                if (document.querySelector(`#mobilephone_${controllId}`) != null && data.MobilePhoneObfuscated && data.MobilePhoneObfuscated.length > 0) {
                                    document.getElementById(`mobilephone_${controllId}`).setAttribute("data-address", data.MobilePhoneObfuscated);
